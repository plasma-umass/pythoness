class Solution:
    
    def atMostNGivenDigitSet(self, digits: List[str], n: int) -> int:
        
        if not digits or n <= 0:
            return 0
        s = str(n)
        k = len(s)
        dp = [0] * (k + 1)
        dp[k] = 1
        for i in range(k - 1, -1, -1):
            for digit in digits:
                if digit < s[i]:
                    dp[i] += len(digits) ** (k - i - 1)
                elif digit == s[i]:
                    dp[i] += dp[i + 1]
        total = sum((len(digits) ** i for i in range(1, k)))
        total += dp[0]
        return total
