class Solution:
    
    def numberOfArrays(self, s: str, k: int) -> int:
        
        MOD = 10 ** 9 + 7
        n = len(s)
        dp = [0] * (n + 1)
        dp[0] = 1
        for i in range(1, n + 1):
            num = 0
            for j in range(i, max(0, i - 10), -1):
                num = int(s[j - 1:i])
                if num > k:
                    break
                if s[j - 1] == '0':
                    continue
                dp[i] = (dp[i] + dp[j - 1]) % MOD
        return dp[n]
