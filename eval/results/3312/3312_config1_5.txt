class Solution:
    
    def gcdValues(self, nums: List[int], queries: List[int]) -> List[int]:
        
        from math import gcd
        n = len(nums)
        gcdPairs = []
        for i in range(n):
            for j in range(i + 1, n):
                gcdPairs.append(gcd(nums[i], nums[j]))
        gcdPairs.sort()
        answer = [gcdPairs[q] for q in queries]
        return answer
