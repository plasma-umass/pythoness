class Solution:
    from typing import List
    
    def firstMissingPositive(self, nums: List[int]) -> int:
        
        n = len(nums)
        for i in range(n):
            while 1 <= nums[i] <= n and nums[nums[i] - 1] != nums[i]:
                # Swap elements to their correct positions
                (nums[nums[i] - 1], nums[i]) = (nums[i], nums[nums[i] - 1])
        for i in range(n):
            if nums[i] != i + 1:
                return i + 1
        return n + 1
