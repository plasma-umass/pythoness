class Solution:
    
    def validSubarraySize(self, nums: List[int], threshold: int) -> int:
        
        n = len(nums)
        for k in range(1, n + 1):
            for i in range(n - k + 1):
                if all((num > threshold / k for num in nums[i:i + k])):
                    return k
        return -1
