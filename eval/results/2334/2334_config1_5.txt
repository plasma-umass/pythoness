class Solution:
    
    def validSubarraySize(self, nums: List[int], threshold: int) -> int:
        
        n = len(nums)
        for k in range(1, n + 1):
            for i in range(n - k + 1):
                valid = True
                for j in range(i, i + k):
                    if nums[j] <= threshold / k:
                        valid = False
                        break
                if valid:
                    return k
        return -1
